// Auf https://www.stellwerksim.de/openedit/edit-bahnhofplan.php in der Console ausführen
(function() {
    let capture = false;
    let regions = [
        'Bern - Lötschberg',
        'Ostschweiz',
        'Tessin',
        'Westschweiz',
        'Zürich und Umgebung',
        'Zentralschweiz',
        'Bourgogne-Franche-Comte',
        'Lombardia'
    ];
    let output = [];
    for (const text of [...$('#liste > tbody > tr > td:nth-child(1)')].map(s => s.innerText))
    {
        const aid = parseInt(text);
        if (isNaN(aid)) {
            capture = regions.includes(text);
        }
        
        if (capture)
            output.push(aid);
    }

    return output;
})()
