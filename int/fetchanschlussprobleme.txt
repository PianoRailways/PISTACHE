// Auf https://www.stellwerksim.de/openedit/edit-bahnhofplan.php in der Console ausführen
// https://www.stellwerksim.de/openedit/edit-bahnhofplan.php?m=plan&aid=AID#AID,AID
(function() {
    let capture = false;
    let regions = [
        'Bern - Lötschberg',
        'Ostschweiz',
        'Tessin',
        'Westschweiz',
        'Zürich und Umgebung',
        'Zentralschweiz',
        'Bourgogne-Franche-Comte',
        'Lombardia',
    ];
    let output = [];
    
    for (const text of [...$('#liste > tbody > tr > td:nth-child(1)')].map(s => s.innerText.trim()))
    {
        const aid = parseInt(text);
        
        if (isNaN(aid)) {
            capture = regions.includes(text);
        } else if (capture) {
            output.push(aid);
        }
    }

    const anzahl = output.length;
    const liste = output.join(',');

    return `Es wurden ${anzahl} Stellwerke gefunden.\n${liste}`;
})()
